width() : lấy độ rộng của thẻ html
    height() : lấy độ cao của thẻ html
    innerWidth() : lấy độ rộng của thẻ html + padding left + padding right
    innerHeight() : lấy độ cao của thẻ html + padding top + padding bottom
    outerWidth() : lấy độ rộng của thẻ html + padding left + padding right + border
    outerHeight() : lấy độ cao của thẻ html + padding top + padding bottom + border
    outerWidth(true) : lấy độ rộng của thẻ html + padding left + padding right + border + margin
    outerHeight(true) : lấy độ cao của thẻ html + padding top + padding bottom + border + margin
