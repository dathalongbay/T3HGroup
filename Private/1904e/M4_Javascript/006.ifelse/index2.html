Câu lệnh rẽ nhanh có nhiều điều kiện

    cú pháp
    if (điều kiện 1) {
        // điều kiện 1 == true thì chạy các câu lệnh ở đây
    } else if (điều kiện 2) {
        // điều kiện 2 == true thì chạy các câu lệnh ở đây
    } else if (điều kiện 3) {
        // điều kiện 3 == true thì chạy các câu lệnh ở đây
    } ...
    else {
        // các trường hợp còn lại
    }
